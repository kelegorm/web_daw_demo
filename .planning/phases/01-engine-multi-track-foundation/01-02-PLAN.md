**Step A: Create TrackFacadeImpl class** (inside engineSingleton.ts, not exported)

```typescript
class TrackFacadeImpl implements TrackFacade {
  #disposed = false;
  #strip: TrackStripGraph;

  constructor(strip: TrackStripGraph) {
    this.#strip = strip;
  }

  get meterSource(): MeterSourceType {
    this.#assertNotDisposed();
    return this.#strip.meterSource;
  }

  setGain(db: number): void {
    this.#assertNotDisposed();
    this.#strip.setTrackVolume(db);
  }

  getGain(): number {
    this.#assertNotDisposed();
    return this.#strip.trackVolume;
  }

  setMute(muted: boolean): void {
    this.#assertNotDisposed();
    this.#strip.setTrackMuted(muted);
  }

  isMuted(): boolean {
    this.#assertNotDisposed();
    return this.#strip.isTrackMuted;
  }

  dispose(): void {
    if (this.#disposed) return;
    this.#disposed = true;
    this.#strip.dispose();
  }

  get isDisposed(): boolean {
    return this.#disposed;
  }

  /** Internal: the raw strip output for bus wiring. NOT exposed on TrackFacade interface. */
  get _stripOutput(): AudioNode {
    return this.#strip.output;
  }

  /** Internal: the raw strip input for device chain wiring. NOT exposed on TrackFacade interface. */
  get _stripInput(): AudioNode {
    return this.#strip.input;
  }

  #assertNotDisposed(): void {
    if (this.#disposed) {
      throw new Error('[TrackFacade] method called on disposed facade');
    }
  }
}
```

**Step B: Replace stub track methods in createEngineInternal()**

Remove the stub implementations of `getTrackFacade`, `createTrackSubgraph`, and `removeTrackSubgraph`. Replace with real implementations:

1. **Track registry:** Already has `const tracks = new Map<string, { facade: TrackFacadeImpl; strip: TrackStripGraph }>();` (or update to use TrackFacadeImpl which holds the strip internally).

   Simplify to: `const tracks = new Map<string, TrackFacadeImpl>();`

2. **createTrackSubgraph(trackId: string): TrackFacade**
   ```
   if (tracks.has(trackId)) {
     throw new Error(`[engine] track already exists: ${trackId}`);
   }
   const strip = createTrackStrip(audioContext);
   const facade = new TrackFacadeImpl(strip);
   strip.output.connect(preLimiterBus);
   tracks.set(trackId, facade);
   return facade;
   ```

3. **removeTrackSubgraph(trackId: string): void**
   ```
   const facade = tracks.get(trackId);
   if (!facade) {
     throw new Error(`[engine] unknown track: ${trackId}`);
   }
   // Disconnect from bus BEFORE disposing nodes (Pitfall 3 from RESEARCH.md)
   facade._stripOutput.disconnect(preLimiterBus);
   facade.dispose();
   tracks.delete(trackId);
   ```

4. **getTrackFacade(trackId: string): TrackFacade**
   ```
   const facade = tracks.get(trackId);
   if (!facade) {
     throw new Error(`[engine] unknown track: ${trackId}`);
   }
   return facade;
   ```
